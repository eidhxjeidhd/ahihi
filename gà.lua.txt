-- Auto Stats System
local function CalculateStatPoints()
   local player = game.Players.LocalPlayer
   local level = player.Data.Level.Value
   local statPoints = level - 1
    
   local meleePoints = math.floor(statPoints * (_G.MeleePercentage or 60) / 100)
   local defensePoints = math.floor(statPoints * (_G.DefensePercentage or 20) / 100)
   local swordPoints = math.floor(statPoints * (_G.SwordPercentage or 20) / 100)
   local gunPoints = math.floor(statPoints * (_G.GunPercentage or 0) / 100)
   local fruitPoints = math.floor(statPoints * (_G.FruitPercentage or 0) / 100)
    
   return meleePoints, defensePoints, swordPoints, gunPoints, fruitPoints
end

local function UpgradeStats()
   local melee, defense, sword, gun, fruit = CalculateStatPoints()
   
   -- Upgrade Melee
   for i = 1, melee do
      game:GetService("ReplicatedStorage").Remotes.CommF_:InvokeServer("AddPoint", "Melee", 1)
   end
   
   -- Upgrade Defense
   for i = 1, defense do
      game:GetService("ReplicatedStorage").Remotes.CommF_:InvokeServer("AddPoint", "Defense", 1)
   end
   
   -- Upgrade Sword
   for i = 1, sword do
      game:GetService("ReplicatedStorage").Remotes.CommF_:InvokeServer("AddPoint", "Sword", 1)
   end
   
   -- Upgrade Gun
   for i = 1, gun do
      game:GetService("ReplicatedStorage").Remotes.CommF_:InvokeServer("AddPoint", "Gun", 1)
   end
   
   -- Upgrade Blox Fruit
   for i = 1, fruit do
      game:GetService("ReplicatedStorage").Remotes.CommF_:InvokeServer("AddPoint", "Demon Fruit", 1)
   end
end
